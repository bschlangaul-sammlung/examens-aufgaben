Gegeben sei der folgende AVL-Baum $T$. Führen Sie auf $T$ folgende
Operationen durch.\index{AVL-Baum}
\footcite{examen:66115:2012:09}

% didaktik.java baum -a 15 9 25 4 10 23 33 2 27  -t -v
\begin{center}
\begin{tikzpicture}[b binaer baum]
\Tree
[.\node[label=0]{15};
  [.\node[label=-1]{9};
    [.\node[label=-1]{4};
      [.\node[label=0]{2}; ]
      \edge[blank]; \node[blank]{};
    ]
    [.\node[label=0]{10}; ]
  ]
  [.\node[label=+1]{25};
    [.\node[label=0]{23}; ]
    [.\node[label=-1]{33};
      [.\node[label=0]{27}; ]
      \edge[blank]; \node[blank]{};
    ]
  ]
]
\end{tikzpicture}
\end{center}

\begin{bAntwort}
Wir führen alle Operationen am Ursprungsbaum $T$ durch und nicht am
veränderten Baum.
\end{bAntwort}
