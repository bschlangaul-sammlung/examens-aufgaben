\section{Aufgabe 2
\index{Kontextfreie Sprache}
\footcite{examen:66115:2013:03}}

Gegeben sei die Grammatik \liGrammatik{variablen={S, A, B, C},
alphabet={a,b}} und

\begin{liProduktionsRegeln}
S -> A B,
S -> C S,
A -> B C,
A -> B B,
A -> a,
B -> A C,
B -> b,
C -> A A,
C -> B A
\end{liProduktionsRegeln}
\liFlaci{Gr46a6j0a}

$L = L(G)$ ist die von G erzeugte Sprache.

\begin{enumerate}
%%
% a)
%%

\item Zeigen Sie, dass $G$ mehrdeutig ist.

\begin{liAntwort}
Das Wort $baab$ kann in zwei verschiedenen Ableitungen hergeleitet
werden:

\begin{enumerate}
\item \liAbleitung{S -> AB -> BCB -> bCB -> bAAB -> baAB -> baaB -> baab}

\item \liAbleitung{S -> CS -> BAS -> bAS -> baS -> baAB -> baaB -> baab}
\end{enumerate}
\end{liAntwort}

%%
% b)
%%

\item Entscheiden Sie mithilfe des Algorithmus von Cocke, Younger und
Kasami (CYK), ob das Wort $w = babaaa$ zur Sprache L gehört. Begründen
Sie Ihre Entscheidung.
\index{CYK-Algorithmus}

\begin{liAntwort}
\begin{tabular}{|c|c|c|c|c|c|}
b     & a     & b    & a    & a    & a \\\hline\hline

B     & A     & B    & A    & A    & A \l6
C     & S     & C    & C    & C \l5
-     & B     & A    & B \l4
A     & C     & A,C \l3
A,C   & B,C,A \l2
A,C,B \l1
\end{tabular}

\liWortNichtInSprache{babaaa}

Das Startsymbol $S$ ist nicht in der Zelle $V(1,5) = \liMenge{A, C, B}$
enthalten.
\end{liAntwort}

%%
% c)
%%

\item Geben Sie eine Ableitung für $w = babaaa$ an.

\begin{liAntwort}
\liAbleitung{A -> BB -> bB -> bAC -> baC -> baAA -> baBCA -> babCA ->
babAAA -> babaAA -> babaaA -> babaaa}
\end{liAntwort}

\end{enumerate}
