\item Geben Sie einen möglichst einfachen regulären Ausdruck für die
Sprache
\liAusdruck[L_1]
{a_1,a_2,\dots,a_n}
{
  n \geq 3, a_i \in \liMenge{a, b}
  \text{ für alle } i = 1,\dots,n
  \text{ und } a_1 \geq a_n
}
an.

\begin{liAntwort}
\texttt{((a(a|b)+b)|(b(a|b)+a))}
\end{liAntwort}

%%
% b)
%%

\item Geben Sie einen möglichst einfachen regulären Ausdruck für die
Sprache
\liAusdruck[L_2]
{w \in \m{a,b}^*}
{
  w
  \text{ enthält genau ein }
  b
  \text{ und ist von ungerader Länge}
}
an.

\begin{liAntwort}
\texttt{(aa)*(b|aba)(aa)*}
\end{liAntwort}
