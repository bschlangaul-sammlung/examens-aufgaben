\chapter{Wiederholungen}

%-----------------------------------------------------------------------
%
%-----------------------------------------------------------------------

\section{Wiederholung mit fester Anzahl}

\begin{minted}{java}
for (int i = 0; i < 3; i++) {
  System.out.println(i + 1);
}
\end{minted}

%-----------------------------------------------------------------------
%
%-----------------------------------------------------------------------

\section{Bedingte Wiederholung}

\begin{minted}{java}
int i = 0;
while (i < 3) {
  System.out.println(i + 1);
  i++;
}
\end{minted}

%-----------------------------------------------------------------------
%
%-----------------------------------------------------------------------

\section{Wiederholung mit Endbedingung}

\begin{minted}{java}
int i = 0;
do {
  System.out.println(i + 1);
  i++;
} while (i < 3);
\end{minted}
