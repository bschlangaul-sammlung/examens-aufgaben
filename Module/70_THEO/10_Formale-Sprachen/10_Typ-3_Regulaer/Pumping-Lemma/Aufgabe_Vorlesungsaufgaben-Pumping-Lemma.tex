\section{Übung zum Pumping-Lemma\footcite[Seite 65]{theo:fs:1}
\index{Pumping-Lemma (Reguläre Sprache)}}

\begin{liExkurs}[Pumping-Lemma für Reguläre Sprachen]
\bPumpingRegulaer
\end{liExkurs}

\begin{enumerate}

%%
%
%%

\item Zeigen Sie, dass die Sprache \bAusdruck{a^n b^m}{n \geq m \geq 1}
nicht regulär ist.

\begin{liAntwort}
$|a^j b^j| \geq j$

$a^j b^j = uvw$
mit $|uv| \leq j$
und $|v| \geq 1$

$\Rightarrow$ in $v$ nur $a$’s

$\Rightarrow$ $uv^0w \notin L$
\end{liAntwort}

%%
%
%%

\item Zeige, dass die Sprache \bAusdruck{a^n b^m}{n > m \geq 1} nicht
regulär ist.

\begin{liAntwort}
$|a^{j+1} b^j| \geq j $

$a^{j+1} b^j = uvw$
mit $|uv| \leq j$
und $|v| \geq 1$

$\Rightarrow$ in $v$ nur $a$’s

$\Rightarrow$ $uv^0w \notin L$
\end{liAntwort}
\end{enumerate}
