\section{Übung\footcite[Seite 34]{theo:fs:2}}

\begin{enumerate}
\item Erstelle eine (deterministische) Grammatik für Palindrome, für die
ein DPDA existiert.

$L = \{ w \$ w^R \, | \, w \in (a|b)^* \}$

\item Wandle diese Grammatik in einen DPDA um.
\end{enumerate}

\begin{itemize}
