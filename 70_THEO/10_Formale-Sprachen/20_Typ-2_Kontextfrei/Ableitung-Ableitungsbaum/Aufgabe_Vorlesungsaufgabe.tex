\item Erstelle eine Ableitung und einen Parsebaum für die nebenstehende
Grammatik für das Wort
\liFlaci{Gpmohr81a}

$V = \m{ S, A, B }$

\liAlphabet{0, 1}

\begin{liProduktionsRegeln}
S -> A1B,
A -> 0A | epsilon,
B -> 0B | 1B | epsilon
\end{liProduktionsRegeln}

$S = S$

\begin{itemize}
\item 10101

\begin{liAntwort}
\liAbleitung{S -> A1B -> 1B -> 10B -> 101B -> 1010B -> 10101B -> 10101}

\begin{center}
\begin{tikzpicture}[li parsetree,level distance=1cm]
\Tree [.S [.A $\varepsilon$ ] 1 [.B 0 [.B 1 [.B 0 [.B 1 [.B $\varepsilon$ ] ] ] ] ] ]
\end{tikzpicture}
\end{center}
\end{liAntwort}

\item 00100

\begin{liAntwort}
\liAbleitung{S -> A1B -> 0A1B -> 00A1B -> 001B -> 0010B -> 00100B -> 00100}

\begin{tikzpicture}[li parsetree,level distance=1cm]
\Tree [.S [.A 0 [.A 0 [.A ] $\varepsilon$ ] ] 1 [.B 0 [.B 0 [.B $\varepsilon$  ] ] ] ]
\end{tikzpicture}
\end{center}
\end{liAntwort}
\end{itemize}

\begin{liAntwort}
